13) Decoradores encadenados

funcion log_llamada(func) {
  print("Decorador: Funcion '" + string(func) + "' fue decorada")
  retornar func
}

funcion medir_tiempo(func) {
  print("Decorador: Midiendo tiempo de ejecucion")
  retornar func
}

@log_llamada
@medir_tiempo
funcion calcular(a, b) {
  retornar a + b
}

sea resultado = calcular(10, 20)
print("Resultado: " + string(resultado))
